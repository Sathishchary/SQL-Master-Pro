WITH top_students AS (
  SELECT user_id, COUNT(*) AS solved,
         SUM(xp) AS total_xp
  FROM   submissions
  WHERE  correct = true
  GROUP BY user_id
)
SELECT u.username, s.solved,
       RANK() OVER (ORDER BY s.total_xp DESC)
FROM   users u JOIN top_students s
       ON u.id = s.user_id
LIMIT 10;
