SELECT u.name,
       COUNT(o.id) AS orders,
       SUM(o.total) AS revenue
FROM   users u
JOIN   orders o
       ON u.id = o.user_id
WHERE  o.created_at >= '2024-01-01'
GROUP BY u.name
ORDER BY revenue DESC
LIMIT  10;
